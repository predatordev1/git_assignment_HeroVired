server {
    listen 80;
    server_name _;  # Accepts requests from any domain/IP

    # Log files
    access_log /var/log/nginx/nodejs-app-access.log;
    error_log /var/log/nginx/nodejs-app-error.log;

    location / {
        proxy_pass http://localhost:3000;  # Change port if your Node.js app uses different port
        proxy_http_version 1.1;

        # Proxy headers
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Timeouts
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;

        # Cache
        proxy_cache_bypass $http_upgrade;
    }

    # Health check endpoint (optional but recommended)
    location /health {
        proxy_pass http://localhost:3000/health;
        access_log off;
    }
}
